user表
	id -- id（-1固定为接收对活动本身评论的虚拟user;-2固定为接收广场消息的虚拟user）
	email -- 邮箱
	phone -- 手机
	pwd -- 密码
	name -- 姓名
	nname -- 昵称(取自nickname)
	sex -- 性别
	birth -- 出生日期
	home -- 家乡

	college -- 学校
	academy	-- 学院
	major -- 专业
	entryy -- 入学年份(取自entry year)
	
	qq -- qq
	wx -- 微信
	rr -- 人人
	fb -- facebook
	tw -- twitter
	
	estate -- 情感状态(取自emotional state)
	hobby -- 兴趣爱好
	tag -- 标签
	sign -- 签名
	
	follow_num -- 关注数
	browse_num -- 浏览数

act表
	a_id -- 活动id
	u_id -- 活动创建的用户id（user表的外码）
	
	a_name -- 活动标题
	deadline -- 报名截止时间
	create_time -- 活动创建时间（系统自动生成）
	start_time -- 开始时间
	end_time -- 结束时间
	extra -- 具体内容
	place -- 活动地点
	max_num -- 最大人数
	join_num -- 已报名人数

	t_id -- 分类（type表的外码）
	a_state -- 活动状态
	browse -- 浏览量/阅读量
	a_like -- 点赞数

source表	-- 指msg中的消息来源

msg表	-- 包括所有的消息（系统，私信，活动评论，广场）
	m_id -- id
	from_u_id -- 发送消息用户的id
	to_u_id -- 接受消息用户的id（若是活动评论，且评论的是活动本身，则值固定为-1）
	m_content -- 消息内容
	unrd -- 0表示已读，1表示未读（对活动本身的评论和发布在广场的消息默认为该值为0）
	m_time -- 发送时间
	s_id -- 消息来源的id
	a_id -- 仅当消息为活动评论时有效，值为活动id，对于非活动评论，该值固定为0
	rp_id -- 仅当消息为广场消息时有效，值为回复的帖子的m_id，对于非广场消息，该值固定位0（该0无效）
	floor -- 指二叉树模型中该消息所位于的层数（0指直接发布在广场的消息，1指回复0的消息，2指回复1的消息，3及以上均默认为2）
